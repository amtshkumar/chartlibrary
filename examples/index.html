D3 Chart Library - Interactive Examples

Bar Chart
const barChart = new BarChart('#bar-chart', {
    width: 800,
    height: 400,
    barColor: '#3498db',
    showValues: true
});

const data = [
    { label: 'A', value: 30 },
    { label: 'B', value: 45 },
    { label: 'C', value: 25 },
    { label: 'D', value: 60 }
];

barChart.setData(data).render();

const lineChart = new LineChart('#line-chart', {

Histogram
const histogram = new Histogram('#histogram', {
    width: 800,
    height: 400,
    bins: 20,
    showDensity: true
});

const data = Array.from({length: 1000}, () => 
    Math.random() * 100
);

histogram.setData(data).render();
